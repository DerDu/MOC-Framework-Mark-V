 43:     /**
 44:      * Formats the path
 45:      * 1. Must start with forward slash
 46:      * 2. Must not end with forward slash
 47:      * 3. Must not have double forward slashes
 48:      *
 49:      * @param *string
 50:      *
 51:      * @return string
 52:      */
 53:     protected function format($path)
 54:     {
 55: 
 56:         //replace back slash with forward
 57:         $path = str_replace('\\', '/', $path);
 58: 
 59:         //replace double forward slash with 1 forward slash
 60:         $path = str_replace('//', '/', $path);
 61: 
 62:         //if there is a last forward slash
 63:         if (substr($path, -1, 1) == '/') {
 64:             //remove it
 65:             $path = substr($path, 0, -1);
 66:         }
 67: 
 68:         //if the path does not start with a foward slash
 69:         //and the path does not have a colon
 70:         //(this is a test for windows)
 71:         if (substr($path, 0, 1) != '/' && !preg_match("/^[A-Za-z]+\:/", $path)) {
 72:             //it's safe to add a slash
 73:             $path = '/'.$path;
 74:         }
 75: 
 76:         return $path;
 77:     }

 92:      * as described on the server. The path
 93:      * given must exist.
 94:      *
 95:      * @param string|null root path
 96:      *
 97:      * @return Eden\System\Path
 98:      */
 99:     public function absolute($root = null)
100:     {
101: 
102:         //argument 1 must be a string or null
103:         Argument::i()->test(1, 'string', 'null');
104: 
105:         //if path is a directory or file
106:         if (is_dir($this->data) || is_file($this->data)) {
107:             return $this;
108:         }
109: 
110:         //if root is null
111:         if (is_null($root)) {
112:             //assume the root is doc root
113:             $root = $_SERVER['DOCUMENT_ROOT'];
114:         }
115: 
116:         //get the absolute path
117:         $absolute = $this->format($this->format($root).$this->data);
118: 
119:         //if absolute is a directory or file
120:         if (is_dir($absolute) || is_file($absolute)) {
121:             $this->data = $absolute;
122:             return $this;
123:         }
124: 
125:         //if we are here then it means that no path was found so we should throw an exception
126:         Exception::i()
127:             ->setMessage(self::ERROR_FULL_PATH_NOT_FOUND)
128:             ->addVariable($this->data)
129:             ->addVariable($absolute)
130:             ->trigger();
131:     }
