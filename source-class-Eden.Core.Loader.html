 10: namespace Eden\Core;
 11: 
 12: /**
 13:  * Handler for class autoloading. Many developers writing
 14:  * object-oriented applications create one PHP source file per-class
 15:  * definition. One of the biggest annoyances is having to write a
 16:  * long list of needed includes at the beginning of each script
 17:  * (one for each class). When a class is not found an Autoload
 18:  * class is used to define how it is found. Eden now conforms to the
 19:  * PSR-0 standard.
 20:  *
 21:  * @vendor  Eden
 22:  * @package Core
 23:  * @author  Christian Blanquera cblanquera@openovate.com
 24:  */
 25: class Loader
 26: {
 27: 
 28:     protected static $instance = null;
 29:     protected $root = array();
 30: 
 31:     /**
 32:      * One of the hard thing about instantiating classes is
 33:      * that design patterns can impose different ways of
 34:      * instantiating a class. The word "new" is not flexible.
 35:      * Authors of classes should be able to control how a class
 36:      * is instantiated, while leaving others using the class
 37:      * oblivious to it. All in all its one less thing to remember
 38:      * for each class call. By default we instantiate classes with
 39:      * this method.
 40:      *
 41:      * @param [mixed[,mixed..]]
 42:      *
 43:      * @return object
 44:      */
 45:     public static function i()
 46:     {
 47: 
 48:         if (is_null(self::$instance)) {
 49:             $class = __CLASS__;
 50:             self::$instance = new $class();
 51:         }
 52: 
 53:         return self::$instance;
 54:     }
 55: 
 56:     /**
 57:      * Adds a root and namespace to search in when a class is not
 58:      * already autoloaded
 59:      *
 60:      * @param string|null
 61:      * @param string|null
 62:      *
 63:      * @return Eden\Core\Loader
 64:      */
 65:     public function addRoot($root = null, $namespace = null)
 66:     {
 67: 
 68:         if ($root === true) {
 69:             $root = realpath(__DIR__.'/../..');
 70:         }
 71: 
 72:         //turn \eden\core\ to eden\core
 73:         $namespace = ltrim($namespace, '\\');
 74:         //turn eden\core to /Eden/Core
 75:         $namespace = ucwords(str_replace('\\', DIRECTORY_SEPARATOR, $namespace));
 76:         //support for legacy Eden_Class
 77:         $namespace = str_replace('_', DIRECTORY_SEPARATOR, $namespace);
 78: 
 79:         $this->root[$root] = $namespace;
 80: 
 81:         return $this;
 82:     }
 83: 
 84:     /**
 85:      * Logically includes a class if not included already.
 86:      *
 87:      * @param *string the class name
 88:      *
 89:      * @return Eden\Core\Loader
 90:      */
 91:     public function load($class)
 92:     {
 93: 
 94:         if (!class_exists($class)) {
 95:             $this->handler($class);
 96:         }
 97: 
 98:         return $this;
 99:     }
100: 
101:     /**
102:      * Logically includes a class if not included already.
103:      *
104:      * @param *string the class name
105:      *
106:      * @return bool
107:      */
108:     public function handler($class)
109:     {
110: 
111:         list( $class, $file, $namespace ) = $this->getMeta($class);
112: 
113:         if ($file && require_once( $file )) {
114:             return true;
115:         }
116: 
117:         return false;
118:     }
119: 
120:     /**
121:      * Determines what meta data we can gather about the class
122:      * including namespace and file location
123:      *
124:      * @param *string the class name
125:      *
126:      * @return array
127:      */
128:     public function getMeta($class)
129:     {
130: 
131:         //With namespacing the full namespace is always passed
132:         //turn \eden\core\ to eden\core
133:         $path = ltrim($class, '\\');
134:         //turn eden\core to /Eden/Core
135:         $path = ucwords(str_replace('\\', DIRECTORY_SEPARATOR, $path));
136:         //support for legacy Eden_Class
137:         $path = str_replace('_', DIRECTORY_SEPARATOR, $path);
138: 
139:         foreach ($this->root as $root => $namespace) {
140:             //by default root+path = file
141:             $file = $root.DIRECTORY_SEPARATOR.$path.'.php';
142: 
143:             //if this is really a file
144:             if (file_exists($file)) {
145:                 return array($class, $file, str_replace('/', '\\', $namespace));
146:             }
147: 
148:             if (!$namespace) {
149:                 continue;
150:             }
151: 
152:             //lets try root+namespace+path = file
153:             $file = $root.DIRECTORY_SEPARATOR.$namespace.DIRECTORY_SEPARATOR.$path.'.php';
154: 
155:             //if this is really a file
156:             if (file_exists($file)) {
157:                 return array($class, $file, str_replace('/', '\\', $namespace));
158:             }
159:         }
160: 
161:         return array($class, null, null);
162:     }
163: 
164:     /**
165:      * Auto registers the auto loader to PHP
166:      *
167:      * @return Eden\Core\Loader
168:      */
169:     public function register()
170:     {
171: 
172:         spl_autoload_register(array($this, 'handler'));
173:         return $this;
174:     }
175: }
