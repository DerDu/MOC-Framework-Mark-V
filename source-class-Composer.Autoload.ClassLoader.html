 20:  *     $loader = new \Composer\Autoload\ClassLoader();
 21:  *
 22:  *     // register classes with namespaces
 23:  *     $loader->add('Symfony\Component', __DIR__.'/component');
 24:  *     $loader->add('Symfony',           __DIR__.'/framework');
 25:  *
 26:  *     // activate the autoloader
 27:  *     $loader->register();
 28:  *
 29:  *     // to enable searching the include path (eg. for PEAR packages)
 30:  *     $loader->setUseIncludePath(true);

 32:  * In this example, if you try to use a class in the Symfony\Component

156:      *
157:      * @param string       $prefix  The prefix/namespace, with trailing '\\'
158:      * @param array|string $paths   The PSR-0 base directories
159:      * @param bool         $prepend Whether to prepend the directories

161:      * @throws \InvalidArgumentException
162:      */
163:     public function addPsr4($prefix, $paths, $prepend = false)
164:     {
165: 
166:         if (!$prefix) {
167:             // Register directories for the root namespace.
168:             if ($prepend) {
169:                 $this->fallbackDirsPsr4 = array_merge(
170:                     (array)$paths,
171:                     $this->fallbackDirsPsr4
172:                 );
173:             } else {
174:                 $this->fallbackDirsPsr4 = array_merge(
175:                     $this->fallbackDirsPsr4,
176:                     (array)$paths
177:                 );
178:             }
179:         } elseif (!isset( $this->prefixDirsPsr4[$prefix] )) {
180:             // Register directories for a new namespace.
181:             $length = strlen($prefix);
182:             if ('\\' !== $prefix[$length - 1]) {
183:                 throw new \InvalidArgumentException("A non-empty PSR-4 prefix must end with a namespace separator.");
184:             }
185:             $this->prefixLengthsPsr4[$prefix[0]][$prefix] = $length;
186:             $this->prefixDirsPsr4[$prefix] = (array)$paths;
187:         } elseif ($prepend) {
188:             // Prepend directories for an already registered namespace.
189:             $this->prefixDirsPsr4[$prefix] = array_merge(
190:                 (array)$paths,
191:                 $this->prefixDirsPsr4[$prefix]
192:             );
193:         } else {
194:             // Append directories for an already registered namespace.
195:             $this->prefixDirsPsr4[$prefix] = array_merge(
196:                 $this->prefixDirsPsr4[$prefix],
197:                 (array)$paths

334:     {
335: 
336:         // work around for PHP 5.3.0 - 5.3.2 https://bugs.php.net/50731
337:         if ('\\' == $class[0]) {
338:             $class = substr($class, 1);
339:         }
340: 
341:         // class map lookup
342:         if (isset( $this->classMap[$class] )) {
343:             return $this->classMap[$class];
344:         }
345:         if ($this->classMapAuthoritative) {
346:             return false;
347:         }
348: 
349:         $file = $this->findFileWithExtension($class, '.php');
350: 
351:         // Search for Hack files if we are running on HHVM
352:         if ($file === null && defined('HHVM_VERSION')) {
353:             $file = $this->findFileWithExtension($class, '.hh');
354:         }
355: 
356:         if ($file === null) {
357:             // Remember that this class does not exist.
358:             return $this->classMap[$class] = false;
359:         }
360: 
361:         return $file;
362:     }

364:     private function findFileWithExtension($class, $ext)
365:     {
366: 
367:         // PSR-4 lookup
368:         $logicalPathPsr4 = strtr($class, '\\', DIRECTORY_SEPARATOR).$ext;
369: 
370:         $first = $class[0];
371:         if (isset( $this->prefixLengthsPsr4[$first] )) {
372:             foreach ($this->prefixLengthsPsr4[$first] as $prefix => $length) {
373:                 if (0 === strpos($class, $prefix)) {
374:                     foreach ($this->prefixDirsPsr4[$prefix] as $dir) {
375:                         if (file_exists($file = $dir.DIRECTORY_SEPARATOR.substr($logicalPathPsr4, $length))) {
376:                             return $file;
377:                         }
378:                     }
379:                 }
380:             }
381:         }
382: 
383:         // PSR-4 fallback dirs
384:         foreach ($this->fallbackDirsPsr4 as $dir) {
385:             if (file_exists($file = $dir.DIRECTORY_SEPARATOR.$logicalPathPsr4)) {
386:                 return $file;
387:             }
388:         }
389: 
390:         // PSR-0 lookup
391:         if (false !== $pos = strrpos($class, '\\')) {
392:             // namespaced class name
393:             $logicalPathPsr0 = substr($logicalPathPsr4, 0, $pos + 1)
394:                 .strtr(substr($logicalPathPsr4, $pos + 1), '_', DIRECTORY_SEPARATOR);
395:         } else {
396:             // PEAR-like class name
397:             $logicalPathPsr0 = strtr($class, '_', DIRECTORY_SEPARATOR).$ext;
398:         }
399: 
400:         if (isset( $this->prefixesPsr0[$first] )) {
401:             foreach ($this->prefixesPsr0[$first] as $prefix => $dirs) {
402:                 if (0 === strpos($class, $prefix)) {
403:                     foreach ($dirs as $dir) {
404:                         if (file_exists($file = $dir.DIRECTORY_SEPARATOR.$logicalPathPsr0)) {
405:                             return $file;
406:                         }
407:                     }
408:                 }
409:             }
410:         }
411: 
412:         // PSR-0 fallback dirs
413:         foreach ($this->fallbackDirsPsr0 as $dir) {
414:             if (file_exists($file = $dir.DIRECTORY_SEPARATOR.$logicalPathPsr0)) {
415:                 return $file;
416:             }
417:         }
418: 
419:         // PSR-0 include paths.
420:         if ($this->useIncludePath && $file = stream_resolve_include_path($logicalPathPsr0)) {
421:             return $file;
422:         }
423:     }
424: }
