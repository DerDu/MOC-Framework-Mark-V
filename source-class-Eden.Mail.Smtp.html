 881:     /**
 882:      * Returns a printable encode version of the body
 883:      *
 884:      * @param string
 885:      * @param int line length
 886:      *
 887:      * @return string
 888:      */
 889:     private function quotedPrintableEncode($input, $line_max = 250)
 890:     {
 891: 
 892:         $hex = array(
 893:             '0',
 894:             '1',
 895:             '2',
 896:             '3',
 897:             '4',
 898:             '5',
 899:             '6',
 900:             '7',
 901:             '8',
 902:             '9',
 903:             'A',
 904:             'B',
 905:             'C',
 906:             'D',
 907:             'E',
 908:             'F'
 909:         );
 910:         $lines = preg_split("/(?:\r\n|\r|\n)/", $input);
 911:         $linebreak = "=0D=0A=\r\n";
 912:         /* the linebreak also counts as characters in the mime_qp_long_line
 913:         * rule of spam-assassin */
 914:         $line_max = $line_max - strlen($linebreak);
 915:         $escape = "=";
 916:         $output = "";
 917:         $cur_conv_line = "";
 918:         $length = 0;
 919:         $whitespace_pos = 0;
 920:         $addtl_chars = 0;
 921: 
 922:         // iterate lines
 923:         for ($j = 0; $j < count($lines); $j++) {
 924:             $line = $lines[$j];
 925:             $linlen = strlen($line);
 926: 
 927:             // iterate chars
 928:             for ($i = 0; $i < $linlen; $i++) {
 929:                 $c = substr($line, $i, 1);
 930:                 $dec = ord($c);
 931: 
 932:                 $length++;
 933: 
 934:                 if ($dec == 32) {
 935:                     // space occurring at end of line, need to encode
 936:                     if (( $i == ( $linlen - 1 ) )) {
 937:                         $c = "=20";
 938:                         $length += 2;
 939:                     }
 940: 
 941:                     $addtl_chars = 0;
 942:                     $whitespace_pos = $i;
 943:                 } else {
 944:                     if (( $dec == 61 ) || ( $dec < 32 ) || ( $dec > 126 )) {
 945:                         $h2 = floor($dec / 16);
 946:                         $h1 = floor($dec % 16);
 947:                         $c = $escape.$hex["$h2"].$hex["$h1"];
 948:                         $length += 2;
 949:                         $addtl_chars += 2;
 950:                     }
 951:                 }
 952: 
 953:                 // length for wordwrap exceeded, get a newline into the text
 954:                 if ($length >= $line_max) {
 955:                     $cur_conv_line .= $c;
 956: 
 957:                     // read only up to the whitespace for the current line
 958:                     $whitesp_diff = $i - $whitespace_pos + $addtl_chars;
 959: 
 960:                     //the text after the whitespace will have to be read
 961:                     // again ( + any additional characters that came into
 962:                     // existence as a result of the encoding process after the whitespace)
 963:                     //
 964:                     // Also, do not start at 0, if there was *no* whitespace in
 965:                     // the whole line
 966:                     if (( $i + $addtl_chars ) > $whitesp_diff) {
 967:                         $output .= substr($cur_conv_line, 0, ( strlen($cur_conv_line) -
 968:                                 $whitesp_diff )).$linebreak;
 969:                         $i = $i - $whitesp_diff + $addtl_chars;
 970:                     } else {
 971:                         $output .= $cur_conv_line.$linebreak;
 972:                     }
 973: 
 974:                     $cur_conv_line = "";
 975:                     $length = 0;
 976:                     $whitespace_pos = 0;
 977:                 } else {
 978:                     // length for wordwrap not reached, continue reading
 979:                     $cur_conv_line .= $c;
 980:                 }
 981:             } // end of for
 982: 
 983:             $length = 0;
 984:             $whitespace_pos = 0;
 985:             $output .= $cur_conv_line;
 986:             $cur_conv_line = "";
 987: 
 988:             if ($j <= count($lines) - 1) {
 989:                 $output .= $linebreak;
 990:             }
 991:         } // end for
 992: 
 993:         return trim($output);
 994:     }
