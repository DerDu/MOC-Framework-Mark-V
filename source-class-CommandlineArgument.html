 80:      *
 81:      * @param  string $to_process
 82:      *
 83:      * @throws BuildException
 84:      * @return array
 85:      */
 86:     public static function translateCommandline($to_process)
 87:     {
 88: 
 89:         if (!$to_process) {
 90:             return array();
 91:         }
 92: 
 93:         // parse with a simple finite state machine
 94: 
 95:         $normal = 0;
 96:         $inQuote = 1;
 97:         $inDoubleQuote = 2;
 98: 
 99:         $state = $normal;
100:         $args = array();
101:         $current = "";
102:         $lastTokenHasBeenQuoted = false;
103: 
104:         $tok = strtok($to_process, "");
105:         $tokens = preg_split('/(["\' ])/', $to_process, -1, PREG_SPLIT_DELIM_CAPTURE);
106:         while (( $nextTok = array_shift($tokens) ) !== null) {
107:             switch ($state) {
108:                 case $inQuote:
109:                     if ("'" == $nextTok) {
110:                         $lastTokenHasBeenQuoted = true;
111:                         $state = $normal;
112:                     } else {
113:                         $current .= $nextTok;
114:                     }
115:                     break;
116:                 case $inDoubleQuote:
117:                     if ("\"" == $nextTok) {
118:                         $lastTokenHasBeenQuoted = true;
119:                         $state = $normal;
120:                     } else {
121:                         $current .= $nextTok;
122:                     }
123:                     break;
124:                 default:
125:                     if ("'" == $nextTok) {
126:                         $state = $inQuote;
127:                     } elseif ("\"" == $nextTok) {
128:                         $state = $inDoubleQuote;
129:                     } elseif (" " == $nextTok) {
130:                         if ($lastTokenHasBeenQuoted || strlen($current) != 0) {
131:                             $args[] = $current;
132:                             $current = "";
133:                         }
134:                     } else {
135:                         $current .= $nextTok;
136:                     }
137:                     $lastTokenHasBeenQuoted = false;
138:                     break;
139:             }
140:         }
141: 
142:         if ($lastTokenHasBeenQuoted || strlen($current) != 0) {
143:             $args[] = $current;
144:         }
145: 
146:         if ($state == $inQuote || $state == $inDoubleQuote) {
147:             throw new BuildException("unbalanced quotes in ".$to_process);
148:         }
149: 
150:         return $args;
151:     }
