 18:  * started to where the program stopped.

 36:      * is instantiated, while leaving others using the class
 37:      * oblivious to it. All in all its one less thing to remember

119:         if (isset( $trace['class'], $trace['function'] )
120:             && $trace['class'] == __CLASS__
121:             && $trace['function'] == 'test'
122:         ) {
123:             //go back one more
124:             $offset = 2;

127:         $trace = debug_backtrace();
128:         $trace = $trace[$offset];

132:         $args = array_shift($types);
133:         $index = array_shift($types) - 1;
134: 
135:         if ($index < 0) {
136:             $index = 0;
137:         }
138: 
139:         //if it's not set then it's good because the default value
140:         //set in the method will be it.
141:         if ($index >= count($args)) {
142:             return $this;
143:         }
144: 
145:         $argument = $args[$index];
146: 
147:         foreach ($types as $i => $type) {
148:             if ($this->isValid($type, $argument)) {
149:                 return $this;

169:     /**
170:      * Validates an argument given the type.
171:      *
172:      * @param *string
173:      * @param *mixed
174:      *
175:      * @return bool
176:      */
177:     protected function isValid($type, $data)
178:     {
179: 
180:         $type = $this->getTypeName($type);
181: 
182:         switch ($type) {
183:             case 'number':
184:                 return is_numeric($data);
185:             case 'int':
186:                 return is_numeric($data) && strpos((string)$data, '.') === false;
187:             case 'float':
188:                 return is_numeric($data) && strpos((string)$data, '.') !== false;
189:             case 'file':
190:                 return is_string($data) && file_exists($data);
191:             case 'folder':
192:                 return is_string($data) && is_dir($data);
193:             case 'email':
194:                 return is_string($data) && $this->isEmail($data);
195:             case 'url':
196:                 return is_string($data) && $this->isUrl($data);
197:             case 'html':
198:                 return is_string($data) && $this->isHtml($data);
199:             case 'cc':
200:                 return ( is_string($data) || is_int($data) ) && $this->isCreditCard($data);
201:             case 'hex':
202:                 return is_string($data) && $this->isHex($data);
203:             case 'alphanum':
204:                 return is_string($data) && $this->alphaNum($data);
205:             case 'alphanumscore':
206:                 return is_string($data) && $this->alphaNumScore($data);
207:             case 'alphanumhyphen':
208:                 return is_string($data) && $this->alphaNumHyphen($data);
209:             case 'alphanumline':
210:                 return is_string($data) && $this->alphaNumLine($data);
211:             default:
212:                 break;
213:         }
214: 
215:         $method = 'is_'.$type;
216:         if (function_exists($method)) {
217:             return $method($data);
218:         }
219: 
220:         if (class_exists($type)) {
221:             return $data instanceof $type;
222:         }
223: 
224:         return true;
225:     }
