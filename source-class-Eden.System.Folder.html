 30:      *
 31:      * @return this
 32:      */
 33:     public function create($chmod = 0755)
 34:     {
 35: 
 36:         //argument 1 must be an integer
 37:         Argument::i()->test(1, 'int');
 38: 
 39:         //if chmod is not and integer or not between 0 and 777
 40:         if (!is_int($chmod) || $chmod < 0 || $chmod > 777) {
 41:             //throw an error
 42:             Exception::i(self::ERROR_CHMOD_IS_INVALID)
 43:                 ->addVariable($this->data)
 44:                 ->trigger();
 45:         }
 46: 
 47:         //if it's not a directory
 48:         if (!is_dir($this->data)) {
 49:             //then make it
 50:             mkdir($this->data, $chmod, true);
 51:         }

 53:         return $this;

164:     {
165: 
166:         //argument test
167:         Argument::i()
